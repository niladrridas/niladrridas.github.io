Project 2: Computer Vision with Python

A computer vision project using Python.


        # Python code for computer vision project
        import cv2

def process_image(image):
    if image is not None:
        # Your computer vision processing logic goes here
        # For a simple example, let's just convert the image to grayscale
        gray_image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        return gray_image
    else:
        print("Error: Unable to load image.")
        return None

def detect_objects(image):
    # Your object detection logic goes here
    # For a simple example, let's use Haarcascades for face detection
    face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')
    faces = face_cascade.detectMultiScale(image, scaleFactor=1.3, minNeighbors=5)

    for (x, y, w, h) in faces:
        cv2.rectangle(image, (x, y), (x+w, y+h), (255, 0, 0), 2)

    return image

def capture_tesla_camera():
    # Use the default camera (index 0). You can change the index if you have multiple cameras.
    cap = cv2.VideoCapture(0)

    while True:
        # Capture frame-by-frame
        ret, frame = cap.read()

        # Process the frame using computer vision techniques
        processed_frame = process_image(frame)

        # Detect objects in the processed frame
        frame_with_objects = detect_objects(processed_frame)

        # Display the resulting frame
        cv2.imshow('Tesla Camera Feed', frame_with_objects)

        # Exit on 'q' key press
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    # Release the camera when the script is closed
    cap.release()

if __name__ == "__main__":
    capture_tesla_camera()
    cv2.destroyAllWindows()
